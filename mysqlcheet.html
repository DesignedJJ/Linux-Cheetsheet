MySQL Basics Cheatsheet

Beginner-friendly: create, read, update, delete, and simple schema changes. Tap a card to expand (no JavaScript).

mysql Connect (CLI) Example: mysql -u root -p ▶
Install client: sudo apt install mysql-client (Debian/Ubuntu) • Local socket: mysql -u root -p • Host/port: -h 127.0.0.1 -P 3306 
# Connect and then choose a database
mysql -u root -p
USE appdb;

# One-liner command
mysql -u root -p -e "SELECT VERSION();"
CREATE DATABASE Create & switch Example: CREATE DATABASE appdb; ▶
Use utf8mb4 to support all Unicode (including emoji).
-- Create a database with good defaults
CREATE DATABASE appdb
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_0900_ai_ci;

-- See databases and switch
SHOW DATABASES;
USE appdb;

-- Show which DB you're in
SELECT DATABASE();
CREATE TABLE Make your first table Example: users(id, email, created_at) ▶
Keep it simple: an auto-increment id, a few columns, and timestamps.
CREATE TABLE users (
  id INT UNSIGNED NOT NULL AUTO_INCREMENT,
  email VARCHAR(255) NOT NULL,
  name  VARCHAR(120) NOT NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  UNIQUE KEY uq_users_email (email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

INSERT Add data Example: INSERT INTO users (...) ▶ 
-- Single row
INSERT INTO users (email, name)
VALUES ('alice@example.com', 'Alice');

-- Multiple rows
INSERT INTO users (email, name) VALUES
  ('bob@example.com', 'Bob'),
  ('cathy@example.com', 'Cathy');
SELECT Read data Example: SELECT * FROM users; ▶
Use WHERE to filter, ORDER BY to sort, LIMIT to cap results.
-- All columns
SELECT * FROM users;

-- Specific columns
SELECT id, email, name FROM users;

-- Filter
SELECT * FROM users
WHERE email LIKE '%@example.com';

-- Sort (newest first) + limit
SELECT id, email, created_at FROM users
ORDER BY created_at DESC
LIMIT 10;

-- Count rows
SELECT COUNT(*) AS total_users FROM users;
UPDATE / DELETE Change or remove Example: UPDATE users SET ... WHERE ... ▶
Always include a WHERE clause to avoid touching every row by accident.
-- Update
UPDATE users
SET name = 'Alice Cooper'
WHERE email = 'alice@example.com';

-- Delete
DELETE FROM users
WHERE id = 3;

-- Safety in dev: require keys in UPDATE/DELETE
SET sql_safe_updates = 1;
ALTER TABLE Common changes Example: ADD/MODIFY/DROP ▶ 
-- Add a column
ALTER TABLE users
ADD COLUMN last_login DATETIME NULL;

-- Change a column type/size
ALTER TABLE users
MODIFY COLUMN name VARCHAR(160) NOT NULL;

-- Rename a column (MySQL 8+)
ALTER TABLE users
RENAME COLUMN name TO full_name;

-- Add / remove an index
CREATE INDEX ix_users_created ON users(created_at);
DROP INDEX ix_users_created ON users;

-- Rename a table
RENAME TABLE users TO app_users;
JOIN Combine tables Example: users ↔ posts ▶
Match related rows using keys (e.g., posts.user_id → users.id).
-- Example post table
CREATE TABLE posts (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id INT UNSIGNED NOT NULL,
  title VARCHAR(200) NOT NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  INDEX (user_id),
  FOREIGN KEY (user_id) REFERENCES users(id)
);

-- Show each user's latest posts
SELECT u.id, u.email, p.title, p.created_at
FROM users u
JOIN posts p ON p.user_id = u.id
ORDER BY p.created_at DESC
LIMIT 10;

-- Include users even if they have no posts
SELECT u.id, u.email, COUNT(p.id) AS post_count
FROM users u
LEFT JOIN posts p ON p.user_id = u.id
GROUP BY u.id, u.email;
INDEX Speed up lookups Example: CREATE INDEX ... ▶
Index columns you search/filter/sort on frequently. Too many indexes can slow inserts/updates.
-- Useful composite index: equality first, then sort/filters
CREATE INDEX ix_posts_user_created ON posts(user_id, created_at);

-- Check the plan
EXPLAIN SELECT * FROM posts
WHERE user_id = 5
ORDER BY created_at DESC
LIMIT 20;
Export / Import Quick backup & CSV Example: mysqldump ... ▶ 
# Dump a database (logical backup)
mysqldump -u root -p appdb > appdb.sql

# Restore into an existing database
mysql -u root -p appdb < appdb.sql

# Export a query to CSV (from CLI)
mysql -u root -p -e "SELECT id,email FROM users" appdb \
  | sed 's/\t/,/g' > users.csv
